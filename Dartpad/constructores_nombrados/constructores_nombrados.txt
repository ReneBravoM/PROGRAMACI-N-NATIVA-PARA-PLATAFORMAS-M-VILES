class Hero {
  String name;
  String power;
  bool asAlive;
  
  Hero({
    required this.name,
    required this.power,
    required this.asAlive
     
  });
  Hero.fromJson(Map<String, dynamic> json)
    : name = json ['name'],
      power= json ['power'].
      isAlive = json ['isAlive'];
  @override
  String toString() {
    return '$name, $power - $(isAlive ? 'Yes :)' : 'Nope :(')';
  }
}
